* ~(pvq) :: (~p^~q)
* ~(p^q) :: (~pv~q)
### Commutativity (Com) ###
* (pvq) :: (qvp)
* (p^q) :: (q^p)

### Associativity (Assoc) ###
* [pv(qvr)] :: [(pvq)vr]
* [p^(q^r)] :: [(p^q)^r]

### Distribution (Dist) ###
* [p^(qvr)] :: [(p^q)v(p^r)]
* [pv(q^r)] :: [(pvq)v(pvr)]
